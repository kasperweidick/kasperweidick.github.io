def read_tab(sub):
    """
        ARGS 
            sub must be one of the sub-tables under api.covid19data.dk
        RETURNS
            Pandas dataframe 
    """
    r = requests.get('https://api.covid19data.dk/' + sub)
    return pd.read_json(r.content)

# To date_time and rename columns
john_hopkins_df['dates'] = pd.to_datetime(john_hopkins_df['dates'])
john_hopkins_df.rename(columns={'country_region': 'country', 'province_state': 'state'}, inplace=True)

# Merge population data for each country to the data frame
john_hopkins_df = pd.merge(john_hopkins_df, population_df, on='country', how='left')
john_hopkins_df.sort_values(['country', 'state', 'dates'], inplace=True)

# New variables
john_hopkins_df['deaths_per_mio'] = john_hopkins_df['deaths'] / (john_hopkins_df['population'] / 1000000)

# Rename US to United States
john_hopkins_df.loc[john_hopkins_df['country'] == 'US', 'country'] = 'United States'

  • Which country has the highest death rate?
  • Which dates has been the worst in terms of deaths world wide?
  • How has deaths developed over time?

# Find the mean death per day for the country with the most deaths
country_max_deaths = john_hopkins_df[john_hopkins_df['deaths'] == john_hopkins_df['deaths'].max()]['country']
country_max_deaths_df = john_hopkins_df[john_hopkins_df['country'] == country_max_deaths.iloc[0]]

# Filter days where there have been confirmed cases
country_max_deaths_df = country_max_deaths_df[country_max_deaths_df['confirmed'] >= 1]
deaths_per_day = country_max_deaths_df['deaths'].max() / len(country_max_deaths_df)

print(f'The country with the most deaths is {country_max_deaths.iloc[0]}, with {deaths_per_day} deaths per day since the first case of COVID19 was confirmed in the country.')
